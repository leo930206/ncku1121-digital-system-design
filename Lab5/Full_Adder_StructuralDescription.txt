module Lab5(a, b, c_in, sum, c_out);
input a, b, c_in;
output sum, c_out;

Half_Adder M1 (a, b, w1, w2);
Half_Adder M2 (c_in, w1, sum, w3);
or (c_out, w2, w3);

endmodule

module Half_Adder(a, b, sum, c_out);
input a, b;
output sum, c_out;

xor (sum, a, b);
and (c_out, a, b);

endmodule
